    \@parboxrestore

    {\csname p@footnote\endcsname\@thefnmark}%
    \color@begingroup
    \uncover#2{\@makeLTRfntext{%
        \rule\z@\footnotesep\ignorespaces#1\@finalstrut\strutbox}}%
    \color@endgroup}}
